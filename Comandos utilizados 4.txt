date
sudo dnf install postfix s-nail -y
sudo systemctl enable postfix
sudo systemctl start postfix
sudo systemctl status postfix

habilita los protocolos y puertos necesarios para el envío de correos:

sudo firewall-cmd --add-service=smtp --permanent
sudo firewall-cmd --add-service=smtps --permanent
sudo firewall-cmd --add-port=587/tcp --permanent

sudo nano /etc/postfix/main.cf

relayhost = [smtp.gmail.com]:587
smtp_sasl_auth_enable = yes
smtp_sasl_security_options = noanonymous
smtp_sasl_password_maps = lmdb:/etc/postfix/sasl_passwd
smtp_use_tls = yes

sudo nano /etc/postfix/sasl_passwd
[smtp.gmail.com]:587 menolmenol00@gmail.com:

guardar conf.

sudo chmod 600 /etc/postfix/sasl_passwd
sudo chown root:root /etc/postfix/sasl_passwd

crear archivo hash
sudo postmap /etc/postfix/sasl_passwd

sudo systemctl restart postfix

echo "Nombre: Victor Amparo
Matricula: 2023-1880" | mail -s "MambruSeFueALaGuerra" os3conadrian@gmail.com

Comandos usados practica 3

date
sudo dnf install cups cups-pdf -y
sudo systemctl start cups
sudo systemctl enable --now cups
sudo systemctl status cups

sudo dnf groupinstall "Development Tools"
sudo dnf install cups-devel git
git clone https://github.com/AOSC-Tracking/cups-pdf.git
cd cups-pdf/src

Compila:
gcc -O2 cups-pdf.c -o cups-pdf -lcups

Instala el backend:
sudo cp cups-pdf /usr/lib/cups/backend/
sudo chmod 755 /usr/lib/cups/backend/cups-pdf

cd ..
sudo cp extra/cups-pdf.conf /etc/cups/
sudo cp extra/CUPS-PDF_opt.ppd /usr/share/cups/model/

Reinicia CUPS y crea la impresora:

sudo systemctl restart cups
sudo lpadmin -p PDF \
  -D "Impresora PDF" \
  -L "Virtual" \
  -v cups-pdf:/ \
  -m CUPS-PDF_opt.ppd \
  -E

sudo firewall-cmd --add-port=631/tcp --permanent
sudo firewall-cmd --reload
sudo nano /etc/cups/cupsd.conf
sudo cupsctl --remote-admin --remote-any --share-printers
sudo systemctl restart cups
clear
